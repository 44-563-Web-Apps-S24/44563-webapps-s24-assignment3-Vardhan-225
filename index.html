Typescript

TypeScript to get a value of an interface property.

    interface MyInterface {
    id: number;
    name: string;
    properties: string[];
  }
  
  const myObject: MyInterface = {
    id: 1,
    name: 'foo',
    properties: ['a', 'b', 'c']
  };
  
  function getValue(value: keyof MyInterface) {
    return myObject[value];
  }
  
  getValue('id'); // 1
  getValue('count')
